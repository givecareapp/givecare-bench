\begin{minipage}[t]{0.48\textwidth}
\textbf{Conversational SMS Delivery}
\begin{itemize}[leftmargin=*, noitemsep, topsep=2pt]
    \item Chunked: 6-8 turns across 2-3 days
    \item Progressive disclosure (not overwhelming)
    \item 24-hour cooldown between domains
    \item Natural language questions
    \item ``Skip'' option always available
    \item \textbf{Completion: 75\%} (6/8 caregivers) vs. $\sim$40\% traditional surveys
\end{itemize}
\end{minipage}
\hfill
\begin{minipage}[t]{0.48\textwidth}
\textbf{Scoring \& Validation}
\begin{itemize}[leftmargin=*, noitemsep, topsep=2pt]
    \item Binary: Yes=100, No=0
    \item Reverse scoring for positive items
    \item Domain score = mean of questions
    \item Overall SDOH = mean of 8 domains
\end{itemize}
\textbf{Validation Required (N=200+):}
\begin{itemize}[leftmargin=*, noitemsep, topsep=2pt]
    \item Convergent validity with CWBS/REACH-II
    \item Test-retest reliability (2-week)
    \item Factor structure (CFA)
\end{itemize}
\end{minipage}
